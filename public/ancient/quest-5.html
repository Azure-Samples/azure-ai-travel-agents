async fetchAvailableTools() {
  const toolsResult = await this.apiService.fetchAvailableTools();
  if (toolsResult) {
    this.tools.set(toolsResult.tools.filter((tool) => tool.selected));
  }
}

  • Filters tools to include only those marked as selected for use in chat interactions.

async sendMessage(event: Event) {
  if ((event as KeyboardEvent).shiftKey) {
    return;
  }

  const messageText = this.userMessage();
  if (!messageText.trim()) return;

  this.messagesBuffer.push({
    role: 'user',
    content: messageText,
    reasoning: [],
    timestamp: new Date(),
  });

  this.userMessage.set('');
  this.isLoading.set(true);
  this.assistantMessageInProgress.set(false);

  await this.apiService.streamChatMessage(
    messageText,
    this.tools().filter((tool) => tool.selected)
  );
}
  • Captures user input and pushes it to the message buffer for display.
  • Clears the input field and sets loading states to indicate that the system is processing.
  • Sends the message to the backend along with the selected tools for processing.
private processAgentEvents(event?: ChatEvent) {
  if (event && event.type === 'metadata') {
    this.currentAgentName.set(event.data?.agentName || null);
    this.agentEventsBuffer.push(event);

    const delta: string = event.data?.delta || '';

    switch (event.event) {
      case 'StartEvent':
        this.updateAndNotifyAgentChatMessageState(delta, {
          metadata: {
            events: this.agentEventsBuffer,
          },
        });
        this.assistantMessageInProgress.set(false);
        break;
      case 'StopEvent':
        this.updateAndNotifyAgentChatMessageState(delta, {
          metadata: {
            events: this.agentEventsBuffer,
          },
        });
        this.assistantMessageInProgress.set(false);
        this.isLoading.set(false);
        break;
      case 'AgentToolCallResult':
        let state = {};
        if (typeof event.data.raw === 'string') {
          state = {
            reasoning: [
              {
                content: event.data.raw,
              },
            ],
          };
        }
        this.updateAndNotifyAgentChatMessageState(delta, state);
        break;
      // Additional cases handle other event types...
    }
  }
}
  • Processes incoming SSE events, updating the state of the current chat message dynamically.
  • Handles different event types, such as StartEvent and StopEvent, to manage the lifecycle of a streaming message.
  • Updates the agentEventsBuffer to maintain a history of events for debugging or display purposes.

async fetchAvailableTools(): Promise<{tools: Tools[]} | void> {
  try {
    const response = await fetch(`${this.apiUrl}/api/tools`, {
      method: 'GET',
      headers: {
        'Content-Type': 'application/json',
      },
    });
    if (!response.ok) {
      const { error } = await response.json();
      return this.handleApiError(
        new Error(error || 'An error occurred while fetching tools'),
        response.status
      );
    }
    return (await response.json());
  } catch (error) {
    return this.handleApiError(error, 0);
  }
}
  • Makes a GET request to fetch the list of available tools.
  • Handles HTTP errors gracefully, providing detailed error messages and status codes.
  • Returns the parsed JSON response for further processing.
async streamChatMessage(message: string, tools: Tools[]) {
  try {
    const response = await fetch(`${this.apiUrl}/api/chat`, {
      method: 'POST',
      headers: {
        'Content-Type': 'application/json',
      },
      body: JSON.stringify({ message, tools }),
    });

    if (!response.ok) {
      const { error } = await response.json();
      return this.handleApiError(
        new Error(error || 'An error occurred'),
        response.status
      );
    }

    const decoder = new TextDecoder('utf-8');

    if (!response.body) {
      throw new Error('Readable stream not supported');
    }

    this.chatStreamState.next({ type: 'START' });

    for await (const chunk of response.body) {
      const value = decoder.decode(chunk, { stream: true });

      const jsonValues = value.trim().split(/\n\n+/);

      for (const jsonValue of jsonValues) {
        try {
          const parsedData = JSON.parse(jsonValue);
          this.chatStreamState.next({
            type: 'MESSAGE',
            event: parsedData,
          });
        } catch (error) {
          console.error('Error parsing JSON chunk:', error);
        }
      }
    }

    this.chatStreamState.next({ type: 'END' });
  } catch (error) {
    this.handleApiError(error, 0);
  }
}
  • Streams chat messages using the fetch API with a POST request.
  • Decodes and processes chunks of data from the server in real time.
  • Updates the chatStreamState observable to notify subscribers of new events or errors.
